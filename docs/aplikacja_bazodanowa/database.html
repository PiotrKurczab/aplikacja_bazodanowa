146    def update_record(self, table, record_id, column_name, new_value):
147        """
148        Updates a record in a table.
149        
150        Args:
151            table (str): The name of the table.
152            record_id (int): The ID of the record to update.
153            column_name (str): The name of the column to update.
154            new_value (str): The new value to set for the column.
155        
156        Raises:
157            ValueError: If column_name is 'id' (primary key), as it cannot be updated.
158        """
159        if column_name.lower() == "id":
160            raise ValueError("Cannot update the primary ID field")
161        column_name = column_name.replace(" ", "_")
162        self.c.execute(f'UPDATE "{table}" SET "{column_name}"=? WHERE id=?', (new_value, record_id))
163        self.conn.commit()
